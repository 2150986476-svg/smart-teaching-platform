    sys_user {
        bigint id PK
        varchar username UK
        varchar password
        varchar real_name
        enum role
        varchar email
        varchar phone
        varchar avatar
        tinyint gender
        varchar department
        varchar class_name
        tinyint status
        tinyint first_login
        datetime last_login_at
        datetime created_at
        datetime updated_at
    }
    course {
        bigint id PK
        bigint teacher_id FK
        varchar name
        text description
        varchar cover_image
        varchar semester
        tinyint status
        varchar class_info
        tinyint is_deleted
        datetime created_at
        datetime updated_at
    }
    course_enrollment {
        bigint id PK
        bigint course_id FK
        bigint student_id FK
        datetime enrolled_at
        tinyint is_active
    }
    course_material {
        bigint id PK
        bigint course_id FK
        varchar file_name
        varchar file_type
        bigint file_size
        varchar file_url
        varchar file_md5
        varchar coze_doc_id
        enum parse_status
        varchar parse_message
        int page_count
        datetime created_at
    }
    coze_knowledge_base {
        bigint id PK
        bigint course_id FK UK
        varchar coze_dataset_id
        varchar coze_dataset_name
        enum status
        varchar fail_reason
        int chunk_size
        datetime synced_at
    }
    ai_assistant_bind {
        bigint id PK
        bigint course_id FK UK
        varchar coze_bot_id
        varchar assistant_name
        varchar assistant_avatar
        varchar welcome_message
        text system_prompt
        decimal temperature
        int max_tokens
        tinyint is_active
    }
    knowledge_point {
        bigint id PK
        bigint course_id FK
        bigint parent_id FK
        varchar name
        text description
        decimal weight
        int sort_order
        int level
    }
    chat_session {
        bigint id PK
        varchar session_id UK
        bigint course_id FK
        bigint student_id FK
        varchar title
        int message_count
        tinyint is_deleted
        datetime created_at
        datetime updated_at
    }
    chat_record {
        bigint id PK
        varchar session_id FK
        bigint course_id FK
        bigint student_id FK
        enum role
        text content
        json metadata
        int token_count
        tinyint rating
        datetime created_at
    }
    quiz_batch {
        bigint id PK
        varchar batch_id UK
        bigint course_id FK
        bigint student_id FK
        enum difficulty
        int question_count
        int correct_count
        int total_time_spent
        decimal score
        enum status
        datetime completed_at
    }
    quiz_record {
        bigint id PK
        varchar batch_id FK
        bigint course_id FK
        bigint student_id FK
        int question_index
        json question_content
        varchar student_answer
        varchar correct_answer
        tinyint is_correct
        tinyint is_flagged
        int time_spent
        enum difficulty
        json knowledge_points
        text explanation
        datetime created_at
    }
    wrong_question_book {
        bigint id PK
        bigint student_id FK
        bigint course_id FK
        bigint quiz_record_id FK
        json question_content
        varchar correct_answer
        varchar student_answer
        enum difficulty
        json knowledge_points
        int wrong_count
        datetime last_wrong_at
        tinyint mastered
        datetime mastered_at
        datetime created_at
    }
    ability_dimension {
        bigint id PK
        varchar code UK
        varchar name
        text description
        varchar icon
        int sort_order
        tinyint is_active
    }
    ability_assessment {
        bigint id PK
        bigint student_id FK
        bigint course_id FK
        varchar dimension_code FK
        decimal score
        date assess_date
        enum source
        datetime created_at
    }
    leaderboard {
        bigint id PK
        bigint course_id FK
        bigint student_id FK
        int total_quiz_count
        int total_questions
        int correct_count
        decimal correct_rate
        int total_chat_count
        decimal total_study_hours
        decimal knowledge_coverage
        decimal average_score
        decimal composite_score
        decimal rank_score
        int rank_position
        int rank_change
        date stat_date
        datetime updated_at
    }
    system_config {
        bigint id PK
        varchar config_key UK
        text config_value
        varchar description
        datetime updated_at
    }

数据表一览

sys_user

用户表 — 教师/学生/助教/管理员

course

课程表 — 课程基本信息

course_enrollment

选课表 — 学生与课程多对多关联

course_material

课程资料表 — 上传的文件信息

coze_knowledge_base

Coze知识库绑定表 — 课程↔知识库

ai_assistant_bind

AI助教绑定表 — 课程↔Coze Bot

knowledge_point

知识点表 — 课程树形知识结构

chat_session

对话会话表 — 问答会话分组

chat_record

对话记录表 — AI问答逐条记录

quiz_batch

答题批次表 — 每次答题为一组

quiz_record

答题记录表 — 每道题作答明细

wrong_question_book

错题本表 — 错题自动收集

ability_dimension

能力维度配置表 — 雷达图维度

ability_assessment

能力评估表 — 评分历史

leaderboard

排行榜表 — 课程排名快照

system_config

系统配置表 — 全局参数
